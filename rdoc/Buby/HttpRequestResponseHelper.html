def self.implant(base)
  return if @implanted
  base.class.instance_eval { include(HttpRequestResponseHelper) }
  @implanted = true
end

def request_headers
  if headers=(@req_split ||= req_str.split(/\r?\n\r?\n/, 2))[0]
    @req_headers ||= headers.split(/\r?\n/).map {|h| h.split(/\s*:\s*/,2)}
  end
end
